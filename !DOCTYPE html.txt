  <script>
    function calcular() {
      let precioCompra = 62.30;
      let precioVenta = 78.00;

      let cajas = document.getElementById('cajas').value || 0;

      document.getElementById('cajasCompra').innerText = cajas;
      document.getElementById('cajasVenta').innerText = cajas;

      let totalCompra = cajas * precioCompra;
      let totalVenta = cajas * precioVenta;
      let ganancia = totalVenta - totalCompra;

      document.getElementById('totalCompra').innerText = 'S/ ' + totalCompra.toFixed(2);
      document.getElementById('totalVenta').innerText = 'S/ ' + totalVenta.toFixed(2);
      document.getElementById('ganancia').innerText = 'Ganancia: S/ ' + ganancia.toFixed(2);
    }
